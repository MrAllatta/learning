Program parsing

Each row of the hs_directory dataset is a school. Schools house one or more programs, which independently enroll students. Programs offer a number of seats available to general education and special education students and may also audition students. Programs can screen for grades and have priorities for enrolling from geographic areas and underrepresented minorities.

classify_families <- function() {
  cols_tbl = tibble(col = names(hs_directory)) %>%
    mutate(
      family= case_when(
        str_detect(col, "grade9geapplicantsperseat\\d+$") ~ "grade9geapplicantsperseat",
        str_detect(col, "^seats9ge\\d$")                  ~ "seats9ge",
        str_detect(col, "^requirement_\\d_\\d+$")        ~ "requirement",
        str_detect(col, "^program\\d+$")                 ~ "program",
        str_detect(col, "^grade9swdapplicatsperseat\\d+$") ~ "grade9swdapplicatsperseat",
        str(col, "^grade9swdseatsperseat\\d+$") ~ "grade9swdseatsperseat",
        str_detect(col, "^grade9swdpriority\\d+$") ~ "grade9swdpriority",
        str_detect(col, "^grade9gepriority\\d+$") ~ "grade9gepriority",
        
        TRUE ~ "other"
      )
    )
    cols_tbl %>% count(family, sort=TRUE)
}
