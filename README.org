** Key Piece: SQL Loader Utility
*=utils/query.py=*
#+begin_src python :tangle utils/query.py
from pathlib import Path
from google.cloud import bigquery

CLIENT = bigquery.Client()
SQL_DIR = Path(__file__).resolve().parent.parent / "sql"

def load_sql(name: str, **params) -> str:
    """Load a .sql file and optionally format with parameters."""
    path = SQL_DIR / name
    sql = path.read_text()
    if params:
        sql = sql.format(**params)
    return sql

def run(name: str, **params):
    """Load SQL file, execute on BigQuery, return DataFrame."""
    sql = load_sql(name, **params)
    return CLIENT.query(sql).to_dataframe()
#+end_src

** Recommendations Beyond Immediate Scope

*** 1. *Authentication*
Set =GOOGLE_APPLICATION_CREDENTIALS= env var or use =gcloud auth application-default login=. Add to your Org file's setup block or =.envrc= (with =direnv=):

*** 2. *Cost Control*
BigQuery charges per bytes scanned. Add a dry-run helper:
#+begin_src python
def dry_run(name: str, **params):
    sql = load_sql(name, **params)
    job_config = bigquery.QueryJobConfig(dry_run=True, use_query_cache=False)
    job = CLIENT.query(sql, job_config=job_config)
    mb = job.total_bytes_processed / 1e6
    print(f"This query will process {mb:.1f} MB")
#+end_src

*** 3. *Parameterization — Use BigQuery's Native Parameters*
=str.format()= is fine for learning but vulnerable to injection. For anything beyond throwaway queries:
#+begin_src python
from google.cloud.bigquery import ScalarQueryParameter, QueryJobConfig

def run_parameterized(name, params: list[ScalarQueryParameter]):
    sql = (SQL_DIR / name).read_text()
    config = QueryJobConfig(query_parameters=params)
    return CLIENT.query(sql, job_config=config).to_dataframe()
#+end_src

*** 4. *SQL Linting/Formatting*
Install =sqlfluff= (=pip install sqlfluff=) and configure for BigQuery dialect:
#+begin_src ini
# .sqlfluff
[sqlfluff]
dialect = bigquery
#+end_src
Then in Emacs use =reformatter.el= or =apheleia= to format on save.

*** 5. *Caching Results Locally*
Avoid re-running expensive queries during iteration:
#+begin_src python
import hashlib, pickle
from pathlib import Path

CACHE_DIR = Path("~/.cache/bq-learn").expanduser()
CACHE_DIR.mkdir(exist_ok=True)

def run_cached(name, **params):
    sql = load_sql(name, **params)
    key = hashlib.sha256(sql.encode()).hexdigest()[:16]
    cache_path = CACHE_DIR / f"{key}.pkl"
    if cache_path.exists():
        return pickle.loads(cache_path.read_bytes())
    df = CLIENT.query(sql).to_dataframe()
    cache_path.write_bytes(pickle.dumps(df))
    return df
#+end_src

*** 8. *Org-mode Tangling as an Alternative*
If you ever /do/ want SQL inline in Org with highlighting, you can use
=#+begin_src sql :tangle ../sql/01-select-basics.sql= — edit with full
SQL highlighting via =C-c '=, and =org-babel-tangle= writes the
files. This keeps a single-source-of-truth notebook while still
producing standalone =.sql= files.
